def del_data_if_exist(group: Group, names: Iterable[str], *, logger: Logger = None):
    logger.debug("exist keys: {}".format(list(group.keys())))
    for name in names:
        if name in group.keys():
            del group[name]
            logger.debug("{} was deleted.".format(name))

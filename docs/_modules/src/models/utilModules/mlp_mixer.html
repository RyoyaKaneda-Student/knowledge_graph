[ドキュメント]    def forward(self, x: torch.Tensor) -> torch.Tensor:
        if self.norm_first:
            _x = self.layer_norm(x)
            _x = self.mlp2(self.activation(self.mlp1(_x)))
            x = x + _x
        else:
            _x = x
            _x = self.mlp2(self.activation(self.mlp1(_x)))
            x = x + self.layer_norm(_x)
        return x




[ドキュメント]class MlpMixer(nn.Module):
    def __init__(self, encoder_layer, num_layers, norm=None):
        super().__init__()
        self.layers = nn.ModuleList([copy.deepcopy(encoder_layer) for i in range(num_layers)])
        self.norm = norm


[ドキュメント]    def forward(self, x: torch.Tensor):
        output = x

        for _layer in self.layers:
            output = _layer(output)

        if self.norm is not None:
            output = self.norm(output)

        return output
